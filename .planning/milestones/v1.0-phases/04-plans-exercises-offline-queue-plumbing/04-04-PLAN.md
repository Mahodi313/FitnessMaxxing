**Block 2 — plan_exercises CRUD cross-user gate**:
```typescript
// =====================================================================
// Phase 4 extension: plan_exercises CRUD cross-user gate
// =====================================================================
{
  // Setup: User A owns a plan + an exercise; insert a plan_exercise via admin.
  const planId = randomUUID();
  const exerciseId = randomUUID();
  const planExerciseId = randomUUID();
  await admin.from('workout_plans').insert({ id: planId, user_id: userA.id, name: 'A Plan' });
  await admin.from('exercises').insert({ id: exerciseId, user_id: userA.id, name: 'A Exercise' });
  await admin.from('plan_exercises').insert({
    id: planExerciseId, plan_id: planId, exercise_id: exerciseId, order_index: 0,
  });

  const clientB = clientFor(sessionB);

  // SELECT: User B should not see plan_exercises belonging to User A's plan.
  const selectAttempt = await clientB.from('plan_exercises').select('*').eq('id', planExerciseId);
  assertEmpty(selectAttempt, 'plan_exercises select: User B cannot read User A plan_exercise');

  // INSERT: User B tries to add a plan_exercise to User A's plan via the FK plan_id.
  const insertAttempt = await clientB.from('plan_exercises').insert({
    id: randomUUID(),
    plan_id: planId,                          // points at User A's plan
    exercise_id: exerciseId,                  // also User A's exercise
    order_index: 1,
  }).select();
  // The RLS WITH CHECK clause uses an EXISTS subquery against workout_plans;
  // since plan_id doesn't match a plan owned by user B, INSERT returns 0 rows.
  assertWriteBlocked(insertAttempt, 'plan_exercises insert: User B cannot insert into User A plan');

  // UPDATE: User B tries to update User A's plan_exercise.
  const updateAttempt = await clientB.from('plan_exercises')
    .update({ order_index: 99 }).eq('id', planExerciseId).select();
  assertWriteBlocked(updateAttempt, 'plan_exercises update: User B cannot update User A plan_exercise');

  // DELETE: User B tries to delete User A's plan_exercise.
  const deleteAttempt = await clientB.from('plan_exercises')
    .delete().eq('id', planExerciseId).select();
  assertWriteBlocked(deleteAttempt, 'plan_exercises delete: User B cannot delete User A plan_exercise');

  // Confirm User A's row is intact (defense-in-depth — the prior asserts could pass
  // for the wrong reason if the row was already gone).
  const confirmIntact = await admin.from('plan_exercises').select('*').eq('id', planExerciseId);
  if (confirmIntact.data?.length !== 1) {
    fail('plan_exercises integrity: User A row was deleted by User B attempt');
  } else {
    pass('plan_exercises integrity: User A row survived all User B write attempts');
  }
}
```

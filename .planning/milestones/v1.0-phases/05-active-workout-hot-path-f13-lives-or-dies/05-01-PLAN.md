```typescript
// app/lib/schemas/sessions.ts
//
// Phase 5: Zod 4 schemas for workout_sessions.
//
// Two schema flavors (mirrors plans.ts):
//   - SessionFormInput: what an RHF form yields (notes only — F12 schema-ready, Phase 7 wires UI)
//   - SessionRowSchema: Zod equivalent of Tables<'workout_sessions'> — fed into
//     SessionRowSchema.parse() at every Supabase response boundary in lib/queries/sessions.ts
//     (Pitfall 8.13 — parse, never cast).
//
// Zod 4 idioms (verified via plans.ts analog):
//   - `error:` parameter on issue locales (`message:` deprecated)
//   - .nullable() at the end of optional chains

import { z } from "zod";

// notes ≤ 500 chars — F12 schema-ready (Phase 7 wires UI)
export const sessionFormSchema = z.object({
  notes: z.string().max(500, { error: "Max 500 tecken" }).nullable().optional(),
});
export type SessionFormInput = z.infer<typeof sessionFormSchema>;

export const sessionRowSchema = z.object({
  id: z.string().uuid(),
  user_id: z.string().uuid(),
  plan_id: z.string().uuid().nullable(),
  started_at: z.string(),                // ISO timestamp (default now() — NOT NULL in schema)
  finished_at: z.string().nullable(),
  notes: z.string().nullable(),
  created_at: z.string().nullable(),
});
export type SessionRow = z.infer<typeof sessionRowSchema>;

// Aliases for downstream consumption (mirrors plans.ts pattern)
export const SessionFormSchema = sessionFormSchema;
export const SessionRowSchema = sessionRowSchema;
```

```typescript
// app/lib/schemas/sets.ts
//
// Phase 5: Zod 4 schemas for exercise_sets.
//
// SetFormSchema is the STRICT validation boundary per CONTEXT.md D-15:
//   - weight_kg: min(0) max(500) multipleOf(0.25)  — PITFALLS §1.5 numeric(6,2) truncation guard
//   - reps:      int() min(1) max(60)              — guards typos like -5 or 9999
//   - set_type:  enum default 'working'            — F17-UI deferred V1.1; V1 always 'working'
//
// Type split: setFormSchema uses z.coerce.number() so its INPUT type is `unknown` and
// OUTPUT type is `number | null`. RHF v7's third generic (TTransformedValues) takes the
// OUTPUT alias so handleSubmit receives the parsed shape. See Phase 4 D-11 precedent
// (commit f8b75b6 — app/app/(app)/plans/[id]/exercise/[planExerciseId]/edit.tsx lines 62–63).

import { z } from "zod";

export const setFormSchema = z.object({
  weight_kg: z.coerce
    .number({ error: "Vikt krävs" })
    .min(0, { error: "Vikt måste vara 0 eller högre" })
    .max(500, { error: "Vikt över 500kg verkar fel — kontrollera" })
    .multipleOf(0.25, { error: "Vikt i steg om 0.25kg" }),
  reps: z.coerce
    .number({ error: "Reps krävs" })
    .int({ error: "Reps måste vara ett heltal" })
    .min(1, { error: "Minst 1 rep" })
    .max(60, { error: "Över 60 reps verkar fel — kontrollera" }),
  set_type: z
    .enum(["working", "warmup", "dropset", "failure"])
    .default("working"),
  rpe: z.coerce.number().nullable().optional(),
  notes: z.string().max(500, { error: "Max 500 tecken" }).nullable().optional(),
});
export type SetFormInput = z.input<typeof setFormSchema>;
export type SetFormOutput = z.output<typeof setFormSchema>;

export const setRowSchema = z.object({
  id: z.string().uuid(),
  session_id: z.string().uuid(),
  exercise_id: z.string().uuid(),
  set_number: z.number().int(),
  reps: z.number().int(),
  weight_kg: z.number(),
  rpe: z.number().nullable(),
  set_type: z.enum(["working", "warmup", "dropset", "failure"]),
  completed_at: z.string().nullable(),
  notes: z.string().nullable(),
});
export type SetRow = z.infer<typeof setRowSchema>;

export const SetFormSchema = setFormSchema;
export const SetRowSchema = setRowSchema;
```

async function runLastValueQuery(client: SupabaseClient<Database>, exerciseId: string, currentSessionId: string, userId: string) {
